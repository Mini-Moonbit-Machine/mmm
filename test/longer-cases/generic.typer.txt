let rec iter: forall ["T"] . (arr: Array[T], len: Int) -> ((T) -> Bool) -> Bool {
  let rec do: (yield: (T) -> Bool) -> Bool {
    let rec next: (i: Int) -> Bool {
      if (LE(Var("i"), Prim(Var("len"), Int(1), Sub, kind=Some(Int)))) then {
        if (App(Var("yield"), [Get(Var("arr"), Var("i"))])) then {
          Apply(Var("next"), [Prim(Var("i"), Int(1), Add, kind=Some(Int)), ])
        } else {
          Bool(false)
        }
      } else {
        Bool(false)
      }
    }
    Apply(Var("next"), [Int(0), ])
  }
  Var(do)
}
let rec each: forall ["T"] . (iter: ((T) -> Bool) -> Bool, f: (T) -> Unit) -> Unit {
  let rec yield: (x: T) -> Bool {
    let _: Unit = Apply(Var("f"), [Var("x"), ])
    Bool(true)
  }
  let rec do: () -> Unit {
    if (App(Var("iter"), [Var("yield")])) then {
      Apply(Var("do"), [])
    } else {
      Unit
    }
  }
  Apply(Var("do"), [])
}
