let rec read_ch_until_newline_or_eof: () -> Int {
  let c: Var(Int) = Apply(Var("read_char"), [])
  if (Eq(Var("c"), Int(10))) then {
    Neg(Int(1))
  } else {
    if (Eq(Var("c"), Int(13))) then {
      Neg(Int(1))
    } else {
      Var(c)
    }
  }
}

let move_right: Int = Int(62)

let rec bf: (len: Int, prog: Array[Int]) -> Unit {
  let ptr: Var(Array[Int]) = Array::make(Int(1), Int(0))
  let cells: Var(Array[Int]) = Array::make(Prim(Int(1024), Int(1024), Mul, kind=Some(Int)), Int(0))
  let next_pc: Var(Array[Int]) = Array::make(Int(1), Int(0))
  let rec skip: (pc: Var(Int), level: Int) -> Var(Int) {
    if (LE(Var("len"), Var("pc"))) then {
      Var(pc)
    } else {
      let i: Var(Int) = Get Var("prog")[Var("pc")]
      if (Eq(Var("i"), Var("loop_start"))) then {
        Apply(Var("skip"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), Prim(Var("level"), Int(1), Add, kind=Some(Int)), ])
      } else {
        if (Eq(Var("i"), Var("loop_end"))) then {
          if (Eq(Var("level"), Int(0))) then {
            Var(pc)
          } else {
            Apply(Var("skip"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), Prim(Var("level"), Int(1), Sub, kind=Some(Int)), ])
          }
        } else {
          Apply(Var("skip"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), Var("level"), ])
        }
      }
    }
  }
  let rec aux: (pc: Var(Int)) -> Var(Int) {
    if (LE(Var("len"), Var("pc"))) then {
      Var(pc)
    } else {
      let i: Var(Int) = Get Var("prog")[Var("pc")]
      if (Eq(Var("i"), Var("move_right"))) then {
        let _: Var(Unit) = Put Var("ptr")[Int(0)] = Prim(Get(Var("ptr"), Int(0)), Int(1), Add, kind=Some(Int))
        Apply(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), ])
      } else {
        if (Eq(Var("i"), Var("move_left"))) then {
          let _: Var(Unit) = Put Var("ptr")[Int(0)] = Prim(Get(Var("ptr"), Int(0)), Int(1), Sub, kind=Some(Int))
          Apply(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), ])
        } else {
          if (Eq(Var("i"), Var("increase"))) then {
            let _: Var(Unit) = Put Var("cells")[Get(Var("ptr"), Int(0))] = Prim(Get(Var("cells"), Get(Var("ptr"), Int(0))), Int(1), Add, kind=Some(Int))
            Apply(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), ])
          } else {
            if (Eq(Var("i"), Var("decrease"))) then {
              let _: Var(Unit) = Put Var("cells")[Get(Var("ptr"), Int(0))] = Prim(Get(Var("cells"), Get(Var("ptr"), Int(0))), Int(1), Sub, kind=Some(Int))
              Apply(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), ])
            } else {
              if (Eq(Var("i"), Var("print"))) then {
                let _: Var(Unit) = Apply(Var("print_char"), [Get(Var("cells"), Get(Var("ptr"), Int(0))), ])
                Apply(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), ])
              } else {
                if (Eq(Var("i"), Var("read"))) then {
                  let _: Var(Unit) = Put Var("cells")[Get(Var("ptr"), Int(0))] = App(Var("read_ch_until_newline_or_eof"), [])
                  Apply(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), ])
                } else {
                  if (Eq(Var("i"), Var("loop_start"))) then {
                    let _: Var(Unit) = Put Var("next_pc")[Int(0)] = Var("pc")
                    let rec lp: () -> Var(Unit) {
                      if (Eq(Get(Var("cells"), Get(Var("ptr"), Int(0))), Int(0))) then {
                        let _: Var(Unit) = Put Var("next_pc")[Int(0)] = App(Var("skip"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int)), Int(0)])
                        Unit
                      } else {
                        let _: Var(Unit) = Put Var("next_pc")[Int(0)] = App(Var("aux"), [Prim(Var("pc"), Int(1), Add, kind=Some(Int))])
                        Apply(Var("lp"), [])
                      }
                    }
                    let _: Var(Unit) = Apply(Var("lp"), [])
                    Apply(Var("aux"), [Prim(Get(Var("next_pc"), Int(0)), Int(1), Add, kind=Some(Int)), ])
                  } else {
                    if (Eq(Var("i"), Var("loop_end"))) then {
                      Var(pc)
                    } else {
                      Var(pc)
                    }
                  }
                }
              }
            }
          }
        }
      }
    }
  }
  let _: Var(Int) = Apply(Var("aux"), [Int(0), ])
  Unit
}
