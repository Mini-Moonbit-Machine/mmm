  let _: Var(Unit) = Put Var("arr")[Var("i")] = Get(Var("arr"), Var("j"))
  let _: Var(Unit) = Put Var("arr")[Var("j")] = Var("tmp")
  Unit
}
let rec partition: (arr: Array[Int], low: Int, high: Int) -> Int {
  let pivot: Var(Int) = Get Var("arr")[Var("high")]
  let i: Var(Array[Int]) = Array::make(Int(1), Prim(Var("low"), Int(1), Sub, kind=Some(Int)))
  let rec l: (j: Var(Int)) -> Var(Unit) {
    if (LE(Var("j"), Prim(Var("high"), Int(1), Sub, kind=Some(Int)))) then {
      let _: Var(Unit) = if (LE(Get(Var("arr"), Var("j")), Var("pivot"))) then {
        let _: Var(Unit) = Put Var("i")[Int(0)] = Prim(Get(Var("i"), Int(0)), Int(1), Add, kind=Some(Int))
        Apply(Var("swap"), [Var("arr"), Get(Var("i"), Int(0)), Var("j"), ])
      } else {
        Unit
      }
      Apply(Var("l"), [Prim(Var("j"), Int(1), Add, kind=Some(Int)), ])
    } else {
      Unit
    }
  }
  let _: Var(Unit) = Apply(Var("l"), [Var("low"), ])
  let _: Var(Unit) = Put Var("i")[Int(0)] = Prim(Get(Var("i"), Int(0)), Int(1), Add, kind=Some(Int))
  let _: Var(Unit) = Apply(Var("swap"), [Var("arr"), Get(Var("i"), Int(0)), Var("high"), ])
  Get Var("i")[Int(0)]
}
let rec quicksort: (arr: Array[Int], low: Int, high: Int) -> Unit {

    let pi: Var(Int) = Apply(Var("partition"), [Var("arr"), Var("low"), Var("high"), ])
    let _: Var(Unit) = Apply(Var("quicksort"), [Var("arr"), Var("low"), Prim(Var("pi"), Int(1), Sub, kind=Some(Int)), ])
    Apply(Var("quicksort"), [Var("arr"), Prim(Var("pi"), Int(1), Add, kind=Some(Int)), Var("high"), ])
  } else {
    Unit
  }
}
