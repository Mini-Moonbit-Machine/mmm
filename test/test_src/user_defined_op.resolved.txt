let rec pow: (x: Int, n: Int) -> Int {
  if (Eq(Var("n"), Int(0))) then {
    Int(1)
  } else {
    Var("x") * App(Var("pow"), [Var("x"), Prim(Var("n"), Int(1), Sub, kind=None)])
  }
}
let rec mod: (x: Int, n: Int) -> Int {
  Var("x") - Prim(Prim(Var("x"), Var("n"), Div, kind=None), Var("n"), Mul, kind=None)
}
