let rec fib: (n: Int) -> Int {
  if (LE(Var("n"), Int(1))) then {
    Var(n)
  } else {
    App(Var("fib"), [Prim(Var("n"), Int(1), Sub, kind=Some(Int))]) + App(Var("fib"), [Prim(Var("n"), Int(2), Sub, kind=Some(Int))])
  }
}
let rec main: () -> Var(Unit) {
  Apply(Var("print_int"), [App(Var("fib"), [Int(30)]), ])
}
Unit
