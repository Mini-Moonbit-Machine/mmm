      CtorApp(Nil, [])
    } else {
      CtorApp(Cons, [Get(Var("arr"), Var("i")), App(Var("aux"), [Prim(Var("i"), Int(1), Add, kind=None)]), ])
    }
  }
  Apply(Var("aux"), [Int(0), ])
}
