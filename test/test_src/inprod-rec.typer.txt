let rec inprod: (v1: Array[Double], v2: Array[Double], i: Int) -> Double {
  if (LE(Int(0), Var("i"))) then {
    Prim(Get(Var("v1"), Var("i")), Get(Var("v2"), Var("i")), Mul, kind=Some(Double)) + App(Var("inprod"), [Var("v1"), Var("v2"), Prim(Var("i"), Int(1), Sub, kind=Some(Int))])
  } else {
    Double(0)
  }
}
