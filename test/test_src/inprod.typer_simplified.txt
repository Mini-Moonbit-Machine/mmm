let rec getx: (v: Tuple([Double, Double, Double])) -> Double {
  let (x: Double, _: Double, _: Double) = Var("v")
  Var(x)
}
let rec gety: (v: Tuple([Double, Double, Double])) -> Double {
  let (_: Double, y: Double, _: Double) = Var("v")
  Var(y)
}
let rec getz: (v: Tuple([Double, Double, Double])) -> Double {
  let (_: Double, _: Double, z: Double) = Var("v")
  Var(z)
}
let rec inprod: (v1: Tuple([Double, Double, Double]), v2: Tuple([Double, Double, Double])) -> Double {

}
